<script type="text/python3">
from browser import document as doc
import math
canvas = doc["guitarchord"]
ctx = canvas.getContext("2d")

# 畫圓函式
def circle(x,y,r):
    ctx.beginPath()
    ctx.arc(x, y, r, 0, math.pi*2, True)
    ctx.fill()
    ctx.closePath()

ctx.beginPath()
ctx.lineWidth = 1
ctx.moveTo(200, 200)
ctx.lineTo(200, 300)
ctx.strokeStyle = "blue"
ctx.stroke()
ctx.closePath()

circle(200, 200, 5)
</script>
